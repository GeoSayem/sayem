Tic Tac Toe Game (Python)

This is a simple console-based Tic Tac Toe game written in Python. It demonstrates game logic, functions, loops, and condition checking. 

# Tic Tac Toe Game in Python
# Author: Sayem Ansari

board = [' ' for _ in range(9)]

def print_board():
    print()
    print(board[0], '|', board[1], '|', board[2])
    print('--+---+--')
    print(board[3], '|', board[4], '|', board[5])
    print('--+---+--')
    print(board[6], '|', board[7], '|', board[8])
    print()

def check_win(player):
    wins = [
        (0,1,2), (3,4,5), (6,7,8),
        (0,3,6), (1,4,7), (2,5,8),
        (0,4,8), (2,4,6)
    ]
    for a, b, c in wins:
        if board[a] == board[b] == board[c] == player:
            return True
    return False

def play_game():
    current_player = 'X'
    turns = 0

    while turns < 9:
        print_board()
        try:
            move = int(input(f"Player {current_player}, choose position (1-9): ")) - 1
        except ValueError:
            print("‚ùå Please enter a number between 1 and 9.")
            continue

        if move < 0 or move > 8:
            print("‚ùå Invalid position. Choose 1-9.")
            continue

        if board[move] != ' ':
            print("‚ùå Position already taken.")
            continue

        board[move] = current_player
        turns += 1

        if check_win(current_player):
            print_board()
            print(f"üéâ Player {current_player} wins!")
            return

        current_player = 'O' if current_player == 'X' else 'X'

    print_board()
    print("ü§ù It's a draw!")

play_game()
